import cv2
import numpy as np
import os
import zipfile
from google.colab.patches import cv2_imshow


# 1. Configurações de caminhos
zip_path = '/content/sel_data.zip'
output_folder = 'resultados'

# Cria a pasta de saída se não existir
if not os.path.exists(output_folder):
    os.makedirs(output_folder)

# 2. Processamento do arquivo ZIP
with zipfile.ZipFile(zip_path, 'r') as z:
    # Filtra apenas arquivos de imagem dentro do zip
    lista_arquivos = [f for f in z.namelist() if f.lower().endswith(('.jpg', '.jpeg', '.png'))]
    print(f"Encontradas {len(lista_arquivos)} imagens. Iniciando processamento...")

    for img_name in lista_arquivos:
        try:
            # Lê a imagem diretamente da memória do ZIP
            with z.open(img_name) as file:
                img_data = np.frombuffer(file.read(), np.uint8)
                img = cv2.imdecode(img_data, cv2.IMREAD_COLOR)

            if img is None:
                continue

            # --- PRÉ-PROCESSAMENTO (Exigido no item 4 do PDF) ---
            # Converte para escala de cinza
            gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)

            # Filtro de Mediana para reduzir ruídos da comida
            gray = cv2.medianBlur(gray, 5)

            # --- TRANSFORMADA DE HOUGH (Item 3 do PDF) ---
            circles = cv2.HoughCircles(
                gray,
                cv2.HOUGH_GRADIENT,
                dp=1.2,          # Resolução do acumulador
                minDist=250,     # Distância mínima entre centros
                param1=120,      # Limiar do Canny
                param2=80,       # Limiar do acumulador (ajuste para detectar mais/menos)
                minRadius=110,    # Raio mínimo do prato
                maxRadius=220     # Raio máximo (0 = sem limite)
            )

            # --- DESENHO DOS RESULTADOS ---
            if circles is not None:
                circles = np.uint16(np.around(circles))
                for j in circles[0, :]:
                    # Desenha o círculo verde (BGR: 0, 255, 0)
                    cv2.circle(img, (j[0], j[1]), j[2], (0, 255, 0), 3)

            # --- SALVAMENTO (Para análise visual do item 5 do PDF) ---
            # Remove o caminho da pasta original para salvar apenas o nome do arquivo
            base_name = os.path.basename(img_name)
            cv2.imwrite(os.path.join(output_folder, f"res_{base_name}"), img)

        except Exception as e:
            print(f"Erro ao processar {img_name}: {e}")

print(f"\nProcessamento concluído! Verifique a pasta '{output_folder}' no menu lateral.")

# Exemplo de visualização de uma imagem processada
try:
    exemplo = cv2.imread(os.path.join(output_folder, os.listdir(output_folder)[0]))
    cv2_imshow(exemplo)
except:
    print("Nenhuma imagem gerada para exibir.")
